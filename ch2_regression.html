DOWNLOAD_ROOT = "https://raw.githubusercontent.com/ageron/handson-ml/master/"
HOUSING_PATH = os.path.join("datasets", "housing")
HOUSING_URL = DOWNLOAD_ROOT + "datasets/housing/housing.tgz"

def fetch_housing_data(housing_url=HOUSING_URL, housing_path=HOUSING_PATH):
  if not os.path.isdir(housing_path):
    os.makedirs(housing_path)
  tgz_path = os.path.join(housing_path, "housing.tgz")
  urllib.request.urlretrieve(housing_url, tgz_path)
  housing_tgz = tarfile.open(tgz_path)
  housing_tgz.extractall(path=housing_path)
  housing_tgz.close()
  
fetch_housing_data()

Looking at this, there are a few little points of concern:

  1. Median Income is not on a dollar scale. It's been scaled somehow. I believe that what we're looking at here is groups of 10k. So a value of 2 means 2*10k or 20,000k.
  2. A few of these have big spikes of data in the last bin. This means that data was capped somehow and the last group actually represents all values greater than a certain value. E.g., in median_house_value we get 500,000+ as the highest item.
